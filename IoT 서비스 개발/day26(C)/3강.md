# break 와 continue



```c
#include <stdio.h>

//break => 반복문 빠져나오기
//continue => 코드 skip

int main()
{
	int i = 0;
	for (i = 1; i <= 10; i++)
	{
		if (i == 5)
		{
			break;
		}
		printf("i = %d\n", i);
	}
	printf("\n");

	for (i = 1; i <= 10; i++)
	{
		if (i == 5)
		{
			continue; // continue 아래의 문자를 실행하지 않고 continue 가 속한 반복문의 첫 문장으로 이동
		}
		printf("i = %d\n", i);
```

* 실습 2차원배열의 합 구하기



```c
#include <stdio.h>

int main()
{
	//int형 myarr[5][5]배열을 선언하고 1부터 25까지 값을 할당하기
	//할당된 배열의 값을 출력하기
	//1 2 3 4 5
	//모든 행과 열의 총합을 저장하고 출력하기
	
	

	
	

	// 배열의 값을 할당하는 for 문
	int myarr[6][6] = { 0 };
	int count = 1;

	for (int row = 0; row < 6; row++)
	{
		for (int col = 0; col < 6; col++)
		{
			if (col == 5 || row == 5) {
				continue;
			}
			myarr[row][col] = count++;
		}
	}

	// 행방향의 합을 구해서 저장하는 for 문
	for (int row = 0; row < 5; row++)
	{
		for (int col = 0; col < 5; col++)
		{
			myarr[row][5] = myarr[row][5] + myarr[row][col];
		}
	}

	// 열방향의 합을 구해서 저장하는 for 문

	for (int col = 0; col < 5; col++)
	{
		for (int row = 0; row < 5; row++)
		{
			myarr[5][col] = myarr[5][col] + myarr[row][col];
		}
	}

	// 배열에 값을 출력하는 for 문
	for (int row = 0; row < 6; row++)
	{
		for (int col = 0; col < 6; col++)
		{
			printf("%d\t", myarr[row][col]);
		}
		printf("\n");
	}

}
```

* 실습 함수 분리해서 사용

```c
#include <stdio.h>

//main(){ UI에 대한 모든작업
//getGrade함수를 호출}
//학점 getGrade(점수)
//{점수를 평가해서 학점을 리턴하는 로직}
char getGrade(int score)
{
	char grade = '#';
	if (score > 100 || score < 0)
	{
		printf("점수 잘못 입력");
	}
	else {
		switch (score / 10)
		{
		case 10:
		case 9:
			printf("A 학점입니다.");
			break;
		case 8:
			printf("B 학점입니다.");
			break;
		case 7:
			printf("C 학점입니다.");
			break;
		case 6:
			printf("D 학점입니다.");
			break;
		default:
			printf("F 학점입니다.");
			break;
		}
	}
	return grade;
}

int main()
{
	int score = 0;
	printf("학점평가\n");
	printf("점수입력 : ");
	scanf_s("%d", &score);
	char grade = getGrade(score);
	printf("%c 학점", grade);
}
```
